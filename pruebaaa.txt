select top 1 Categoria.Nombre
 from Categoria 
 inner join Producto 
 on Categoria.CodigoCategoria=Producto.CodigoCategoria
   inner join Venta 
   on Producto.CodigoProducto=Venta.CodigoProducto order by Fecha Desc;
